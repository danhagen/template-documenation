%% Title Page ith Horizon Logo
\newcommand\horizontitlepage[1][wTOC]{
    \clearpage
    \maketitle
    \IfStrEqCase{#1}{%
        {wTOC}{
            \vfill
            \tableofcontents
            \thispagestyle{empty}
        }%
        {woTOC}{
            \thispagestyle{empty}
            \pagebreak 
            \tableofcontents
        }%
        {noTOC}{
            \thispagestyle{empty}
        }%
    }[%
        \PackageError{horizontitlepage}{Invalid optional argument!}{Either use 'wTOC' (default) to include the table of contents on the title page, 'woTOC' to place the table of contents on a new page, or 'noTOC' to only include the title page (no table of contents).}
    ]%
    \pagebreak
}
